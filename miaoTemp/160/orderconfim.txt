        var proceed = false

        if (CSS.supports('padding-bottom: env(safe-area-inset-bottom)')) {
            div.style.paddingBottom = 'env(safe-area-inset-bottom)'
            proceed = true
        } else if (CSS.supports('padding-bottom: constant(safe-area-inset-bottom)')) {
            div.style.paddingBottom = 'constant(safe-area-inset-bottom)'
            proceed = true
        }
        if (proceed) {
            document.body.appendChild(div)
            var calculatedPadding = parseInt(window.getComputedStyle(div).paddingBottom)
            document.body.removeChild(div)
            if (calculatedPadding > 0) {
                return true
            }
        }
        return false

function middleLoading(type=2){
  $("#middleLoading").show()
  setTimeout(function(){
    var curUrl = window.location.href; // 当前页面地址
    var ishowSourceType = curUrl.indexOf('sourceType') //地址栏是否有sourceType
    var newCurUrl = curUrl.slice(0,parseInt(ishowSourceType)-1)
    var urls = type === 1 ? "" : ""
    var referrers = getUrlParam('referrer') || ''
    if(referrers){
      window.location.href = curUrl
    }else if(ishowSourceType > -1){
      window.location.href = urls+'&referrer='+encodeURIComponent(newCurUrl)
    }else{
      window.location.href = urls+'&referrer='+encodeURIComponent(curUrl)
    }
  }, 1000);
}

function getImageCode() {

    url : "//gate.91160.com/guahao/v1/order/image_data?unit_id=" + unit_id,

    },

  })
}

function showPrompt(title, msg, duration, args, icon){
    var title = title ? title : "", msg = msg ? msg : "", duration = duration ? duration : 3000, args;
    var event_0 = args[0].events ? args[0].events : function(){}, event_1 = typeof args[1] != 'undefined' ? args[1].events : function(){};
    function reset(){
        $(".prompt").removeClass("show");
        $(".prompt-box").removeClass("slipin");
        $(".prompt-icon").remove();
        $(".prompt-title, .prompt-msg, .prompt-btn").empty();
        $(".prompt-btn").unbind();
        $(".prompt-btn").removeClass("half");
        $(".prompt-box, .prompt-msg").removeClass("style");
    }
    if (icon) {
        $('<i class="iconfont prompt-icon" style="font-size: ' + icon.fontSize + ';background-color: ' + icon.bgColor + '">' + icon.icon + '</i>').insertBefore($(".prompt-main"))
    }
    $(".prompt-title").html(title);
    $(".prompt-msg").html(msg);
    if(args.length > 1){
        $(".prompt-btn").addClass("half");
    }
    if(args.length){
        for(var i = 0; i < args.length; i++){
            var curbtn = $(".prompt-btn").eq(i);
            curbtn.text(args[i].name);
            curbtn.attr("href", args[i].address ? args[i].address : "javascript:void(0);");
            if (args[i].color) {
                curbtn.css('cssText','color:' + args[i].color+" !important");
            } else {
                curbtn.css('cssText', '');
            }
            curbtn.one("click", function(){
                var action = $(this).index() ? event_1 : event_0;
                if(typeof action == 'function') action();
                reset();
            });
        }
    }
    $(".prompt").addClass("show");
    $(".prompt-box").addClass("slipin");
    $(".prompt-msg").css({"maxHeight": $(window).height()*0.5+"px"});
    $(".prompt-box").css({"top": ($(window).height() - $(".prompt-box").height())/2+"px"});
    if(duration != "sync" && duration != "syncs"){
        setTimeout(function(){
            reset();
        }, duration);
    }else if(duration == "sync"){
        $(".prompt").one("click", function(){
            reset();
        });
    }
}

    function getBafyRecord(cb) {
        $.ajax({
            url : "//wechatgate.91160.com/guahao/v1/his/has_report",
            timeout : 20000,
            type : 'get',
            dataType : 'json',
            data: {
                unit_id,
                dep_id,

                member_id: 134979053,
                user_key,
                cid: 23 || 16
            },

                    const has_report = res.data.has_report

                    cb && cb(has_report, price)

                }
                cb && cb()
            },
            fail:function(res) {
                // res.error_msg && showTips(res.error_msg)
                cb && cb()
            }
        })
    }
